"""### **QUESTION 8: Write a grading system using if-elif-else.**"""

def calculate_grade(score):
    # Check for invalid scores first
    if score > 100 or score < 0:
        return "Invalid Score! Please enter a value between 0 and 100."

    # Evaluate valid scores top-to-bottom
    elif score >= 90:
        return "A"
    elif score >= 80:
        return "B"
    elif score >= 70:
        return "C"
    elif score >= 60:
        return "D"
    else:
        return "F"

# Example Usage:
try:
    user_input = input("Enter the student's score (0-100): ")
    student_score = float(user_input)

    grade = calculate_grade(student_score)
    print(f"Final Grade: {grade}")

except ValueError:
    print("Error: Please enter a valid number.")

name = input("Enter the student's name: ")

try:
    score = float(input(f"Enter {name}'s score (0-100): "))

    if not (0 <= score <= 100):
        print("Error: Score must be between 0 and 100.")
    else:
        # Determine the grade
        if score >= 90:
            grade = 'A'
        elif score >= 80:
            grade = 'B'
        elif score >= 70:
            grade = 'C'
        elif score >= 60:
            grade = 'D'
        else:
            grade = 'F'

        # Determine pass/fail (assuming 60 is the passing score)
        status = "Pass" if score >= 60 else "Fail"

        print(f"\n--- Result for {name} ---")
        print(f"Score: {score:.2f}")
        print(f"Grade: {grade}")
        print(f"Status: {status}")
except ValueError:
    print("Error: Invalid input for score. Please enter a number.")
